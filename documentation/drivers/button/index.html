Simple interface that makes it possible to register clicks through physical input.

How To Connect

Install the library with: go get github.com/hybridgroup/gobot && go install github.com/hybridgroup/platforms/gpio

For Firmata:

package main

import (
  "github.com/hybridgroup/gobot"
  "github.com/hybridgroup/gobot/platforms/firmata"
  "github.com/hybridgroup/gobot/platforms/gpio"
)

func main() {
  gbot := gobot.NewGobot()

  firmataAdaptor := firmata.NewFirmataAdaptor("myFirmata", "/dev/ttyACM0")

  button := gpio.NewButtonDriver(firmataAdaptor, "myButton", "2")
  led := gpio.NewLedDriver(firmataAdaptor, "myLed", "13")

How To Use

This example detects when a button has been pushed, it turns on and off a LED

package main

import (
  "github.com/hybridgroup/gobot"
  "github.com/hybridgroup/gobot/platforms/firmata"
  "github.com/hybridgroup/gobot/platforms/gpio"
)

func main() {
  gbot := gobot.NewGobot()

  firmataAdaptor := firmata.NewFirmataAdaptor("myFirmata", "/dev/ttyACM0")

  button := gpio.NewButtonDriver(firmataAdaptor, "myButton", "2")
  led := gpio.NewLedDriver(firmataAdaptor, "myLed", "13")

  work := func() {
    gobot.On(button.Events["push"], func(data interface{}) {
      led.On()
    })
    gobot.On(button.Events["release"], func(data interface{}) {
      led.Off()
    })
  }

  gbot.Robots = append(gbot.Robots,
    gobot.NewRobot("buttonBot", []gobot.Connection{firmataAdaptor}, []gobot.Device{button, led}, work),
  )

  gbot.Start()
}
